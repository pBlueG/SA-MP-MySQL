set(BOOST_STATIC_STRING_TESTS_FILES
        CMakeLists.txt
        Jamfile
        constexpr_tests.hpp
        compile_fail.hpp
        static_string.cpp

# of the root project directory.
# See: BOOST_STATIC_STRING_UNIT_TEST_LIBRARIES

add_test(NAME boost_static_string_tests COMMAND boost_static_string_tests)
